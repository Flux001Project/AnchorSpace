```ts
// src/persistence/dayBook.ts

export interface PersistedRunEvent {
  /** Epoch ms. */
  t: number;
  /** Tool name or activity tag. */
  type: string;
  /** Short human-readable summary. */
  summary: string;
}

export interface PersistedRun {
  /** Composite key: `${runId}::${sessionKey}`. Unique. */
  key: string;
  runId: string;
  sessionKey: string;
  /** Stable display name e.g. "Flux", "Sub-1". */
  agentId: string;
  /** Free-text session label (often the human-readable session name). */
  sessionLabel: string;
  /** Epoch ms. */
  startedAt: number;
  /** Epoch ms. null while live. */
  endedAt: number | null;
  /** Total tool calls observed during this run. */
  toolCallCount: number;
  /** Accumulated idle ms. */
  idleMs: number;
  /** Accumulated working ms. */
  workingMs: number;
  /** Last N events (capped per-run by RECENT_EVENTS_PER_RUN). */
  recentEvents: PersistedRunEvent[];
}

export interface DayBookStore {
  /** Sorted by startedAt DESC (newest first). */
  runs: PersistedRun[];
  version: number;
}

/** Schema version. Bump when PersistedRun shape changes. */
export const DAYBOOK_VERSION: 1;

/** Rolling-window cap. Oldest entries evict beyond this. */
export const MAX_RUNS: 5000;

/** Per-run cap on `recentEvents` array length. */
export const RECENT_EVENTS_PER_RUN: 20;

/** localStorage key. */
export const DAYBOOK_STORAGE_KEY: "anchorspace.dayBook.v1";

/** Read store from localStorage. Returns empty store on miss/parse error. */
export function loadDayBook(): DayBookStore;

/** Persist store to localStorage. Silently no-ops when window is undefined. */
export function saveDayBook(store: DayBookStore): void;

/**
 * Insert or replace a run by `key`. Maintains startedAt-DESC sort.
 * Enforces MAX_RUNS rolling window: if exceeded, evicts the oldest
 * (smallest startedAt) entries until size ≤ MAX_RUNS.
 * Returns a NEW store — does not mutate the input.
 */
export function upsertRun(store: DayBookStore, run: PersistedRun): DayBookStore;

/**
 * Append an event to a run's `recentEvents`. Caps length at RECENT_EVENTS_PER_RUN
 * by dropping the OLDEST event (FIFO). If the run does not exist in the store,
 * returns the store unchanged. Does NOT mutate the input — returns a new store.
 */
export function appendEvent(
  store: DayBookStore,
  key: string,
  event: PersistedRunEvent
): DayBookStore;

/**
 * Filter to runs whose startedAt falls within [dayStart, dayStart + 24h).
 * `dayStart` should be a local-midnight epoch ms. Returns runs sorted
 * startedAt-DESC.
 */
export function runsForDay(store: DayBookStore, dayStart: number): PersistedRun[];

/**
 * Compute today's local-midnight as epoch ms. Pure helper exposed for tests
 * and for the panel to derive its filter window.
 */
export function localMidnight(now: number): number;

/** Reset the store to empty. Used by tests and a future "clear day" affordance. */
export function clearDayBook(): void;
```

### Example 3 — upsert maintains DESC order and replaces by key
```ts
const empty: DayBookStore = { runs: [], version: 1 };
const a: PersistedRun = { key: "a", runId: "a", sessionKey: "s", agentId: "Flux", sessionLabel: "m", startedAt: 100, endedAt: null, toolCallCount: 0, idleMs: 0, workingMs: 0, recentEvents: [] };
const b: PersistedRun = { ...a, key: "b", runId: "b", startedAt: 200 };
const c: PersistedRun = { ...a, key: "c", runId: "c", startedAt: 150 };
const aUpdated: PersistedRun = { ...a, toolCallCount: 7 };

const s1 = upsertRun(empty, a);
const s2 = upsertRun(s1, b);
const s3 = upsertRun(s2, c);
const s4 = upsertRun(s3, aUpdated);

// expect (DESC by startedAt):
s4.runs.map(r => r.key);                          // ["b", "c", "a"]
s4.runs.find(r => r.key === "a")!.toolCallCount;  // 7
s4.runs.length;                                    // 3 (no duplicate "a")
```

### Example 4 — rolling-window eviction at MAX_RUNS

```ts
for (let i = 0; i < 5005; i++) {

// expect:
store.runs.length;                          // 5000 (MAX_RUNS)
store.runs[0].key;                          // "k5004" (newest, head of DESC list)

```

### Example 5 — appendEvent caps recentEvents and is FIFO
```ts
const r: PersistedRun = {

  recentEvents: [],

let store: DayBookStore = upsertRun({ runs: [], version: 1 }, r);

got.recentEvents[0].summary;         // "e5"
got.recentEvents[19].summary;        // "e24"
```
