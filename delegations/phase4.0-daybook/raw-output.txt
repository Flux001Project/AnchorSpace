/**
 * Insert or replace a run by `key`. Maintains startedAt-DESC sort.
 * Enforces MAX_RUNS rolling window: if exceeded, evicts the oldest
 * (smallest startedAt) entries until size ≤ MAX_RUNS.
 * Returns a NEW store — does not mutate the input.
 */
export function upsertRun(store: DayBookStore, run: PersistedRun): DayBookS[8D[K
DayBookStore {
  const newRuns = [...store.runs];
  const existingIndex = newRuns.findIndex(r => r.key === run.key);
  if (existingIndex !== -1) {
    newRuns[existingIndex] = run;
  } else {
    newRuns.push(run);
  }

  // Sort DESC by startedAt
  newRuns.sort((a, b) => b.startedAt - a.startedAt);

  // Trim to MAX_RUNS
  if (newRuns.length > MAX_RUNS) {
    newRuns.splice(MAX_RUNS);
  }

  return { runs: newRuns, version: store.version };
}

/**
 * Append an event to a run's `recentEvents`. Caps length at RECENT_EVENTS_[14D[K
RECENT_EVENTS_PER_RUN
 * by dropping the OLDEST event (FIFO). If the run does not exist in the st[2D[K
store,
 * returns the store unchanged. Does NOT mutate the input — returns a new s[1D[K
store.
 */
export function appendEvent(
  store: DayBookStore,
  key: string,
  event: PersistedRunEvent
): DayBookStore {
  const newRuns = [...store.runs];
  const runIndex = newRuns.findIndex(r => r.key === key);
  if (runIndex === -1) return store;

  const run = { ...newRuns[runIndex] };
  run.recentEvents = [...run.recentEvents, event];
  if (run.recentEvents.length > RECENT_EVENTS_PER_RUN) {
    run.recentEvents = run.recentEvents.slice(-RECENT_EVENTS_PER_RUN);
  }

  newRuns[runIndex] = run;
  return { runs: newRuns, version: store.version };
}
